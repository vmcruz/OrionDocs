var MostachoDB = function(idbName, idbVersion) {
    if(!idbName)
        return undefined;
    this.idbName = idbName;
    this.idbVersion = idbVersion;
    this.idbConnection = undefined;
    this.tableConfig = [];
    this.tables = [];
    this.fakeIDs = [];
    this.triggers = {};
